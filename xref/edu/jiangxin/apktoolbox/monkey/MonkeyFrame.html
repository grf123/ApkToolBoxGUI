588 	/**
589 	 * 获取设备列表
590 	 * 
591 	 * @param cmd
592 	 */
593 	public void getDevices(String cmd) {
594 
595 		logger.info("获取设备列表......开始");
596 
597 		try {
598 			Process process = Runtime.getRuntime().exec(cmd);
599 
600 			BufferedReader reader = new BufferedReader(new InputStreamReader(process.getInputStream()));
601 
602 			String line = null;
603 
604 			while ((line = reader.readLine()) != null) {
605 
606 				if (line.indexOf("List") < 0 && line.length() != 0) {
607 
608 					line = line.substring(0, line.length() - 7);
609 
610 					logger.info("获取的设备名称：" + line);
611 
612 					comboBoxDevices.addItem(line);
613 
614 				}
615 
616 			}
617 
618 			// process.waitFor();
619 
620 			process.getOutputStream().close(); // 不要忘记了一定要关
621 
622 		} catch (Exception e) {
623 
624 			logger.error("获取设备列表异常", e);
625 
626 		}
627 
628 		logger.info("获取设备列表......结束");
629 
630 	}
